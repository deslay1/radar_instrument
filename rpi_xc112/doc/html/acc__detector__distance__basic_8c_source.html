41 acc_detector_distance_basic_handle_t acc_detector_distance_basic_create(acc_sensor_id_t sensor_id, float range_start, float range_length)
42 {
43  acc_service_configuration_t envelope_configuration = acc_service_envelope_configuration_create();
44 
45  if (envelope_configuration == NULL)
46  {
47  ACC_LOG_ERROR("Distance basic detector not possible to allocate");
48  return NULL;
49  }
50 
51  acc_service_profile_set(envelope_configuration, ACC_SERVICE_PROFILE_2);
52  acc_service_repetition_mode_on_demand_set(envelope_configuration);
53  acc_service_requested_start_set(envelope_configuration, range_start);
54  acc_service_requested_length_set(envelope_configuration, range_length);
55  acc_service_sensor_set(envelope_configuration, sensor_id);
56  acc_service_receiver_gain_set(envelope_configuration, 0.82f);
57  acc_service_hw_accelerated_average_samples_set(envelope_configuration, 7);
58 
59  acc_detector_distance_basic_handle_internal_t *detector_handle = acc_os_mem_alloc(sizeof(*detector_handle));
60 
61  if (detector_handle != NULL)
62  {
63  detector_handle->magic_number = MAGIC_NUMBER;
64  detector_handle->envelope_handle = acc_service_create(envelope_configuration);
65 
66  if (detector_handle->envelope_handle != NULL)
67  {
68  acc_service_envelope_metadata_t envelope_metadata;
69  acc_service_envelope_get_metadata(detector_handle->envelope_handle, &envelope_metadata);
70 
71  detector_handle->envelope_data_length = envelope_metadata.data_length;
72  detector_handle->envelope_data = acc_os_mem_alloc(sizeof(*detector_handle->envelope_data) * detector_handle->envelope_data_length);
73  detector_handle->distance_offset = envelope_metadata.start_m;
74  detector_handle->distance_slope = envelope_metadata.length_m / (float)(envelope_metadata.data_length - 1);
75 
76  if (detector_handle->envelope_data != NULL)
77  {
78  acc_service_activate(detector_handle->envelope_handle);
79  }
80  else
81  {
82  acc_service_destroy(&detector_handle->envelope_handle);
83  acc_os_mem_free(detector_handle);
84  detector_handle = NULL;
85  ACC_LOG_ERROR("Distance basic detector not possible to allocate");
86  }
87  }
88  else
89  {
90  acc_os_mem_free(detector_handle);
91  detector_handle = NULL;
92  ACC_LOG_ERROR("Distance basic detector not possible to create");
93  }
94  }
95  else
96  {
97  detector_handle = NULL;
98  ACC_LOG_ERROR("Distance basic detector not possible to allocate");
99  }
100 
101  acc_service_envelope_configuration_destroy(&envelope_configuration);
102 
103  return (acc_detector_distance_basic_handle_t)detector_handle;
104 }

123 acc_detector_distance_basic_reflection_t acc_detector_distance_basic_get_reflection(acc_detector_distance_basic_handle_t handle)
124 {
125  acc_detector_distance_basic_reflection_t reflection;
126  reflection.distance = 0.0f;
127  reflection.amplitude = 0.0f;
128 
129  if (handle_valid(handle))
130  {
131  uint16_t *envelope_data = handle->envelope_data;
132  uint16_t envelope_data_length = handle->envelope_data_length;
133 
134  acc_service_envelope_result_info_t result_info;
135  acc_service_envelope_get_next(handle->envelope_handle, envelope_data, envelope_data_length, &result_info);
136 
137  uint16_t reflection_index = get_reflection_index(envelope_data, envelope_data_length);
138 
139  reflection.amplitude = envelope_data[reflection_index];
140  reflection.distance = INDEX_TO_DISTANCE_CONVERSION(reflection_index, handle->distance_slope, handle->distance_offset);
141  }
142 
143  return reflection;
144 }

151 {
152  bool valid = true;
153 
154  if ((detector_handle == NULL) || (detector_handle->magic_number != MAGIC_NUMBER))

157  valid = false;
158  }
159 
160  return valid;
161 }

164 uint16_t get_reflection_index(uint16_t *data, uint16_t data_length)
165 {
166  uint16_t reflection_amplitude = 0;
167  uint16_t reflection_index = 0;
168  for (uint16_t i = 0; i < data_length; i++)
169  {
170  if (data[i] > reflection_amplitude)
171  {
172  reflection_amplitude = data[i];
173  reflection_index = i;
174  }
175  }
176 
177  return reflection_index;
178 }
